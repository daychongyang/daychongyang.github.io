# 什么是模块?

模块是一组代码的集合，可以使用其他模块，也可以被其他模块使用。

# 什么是模块化?

代码分割

# 为什么需要模块化?

  1. 减少命名空间污染
  2. 高可维护性
  3. 高复用性
  4. 按需加载

# 模块化进化过程

# global function 模式

/** 污染全局命名空间, 易引起命名冲突, 模块依赖关系不清晰 */
var name = "xiaoming";

function set(newName) {
  name = newName;
}

# namespace 模式

/** 减少了全局变量, 减少了命名冲突, 外部模块可直接修改模块内部数据 */
var module1 = {
  name: "xiaoming",
  set: function(name) {
    this.name = name;
  },
};

# IIFE 模式

/** 数据私有, 只对外保留接口 */
(function(window, dependModule) {
  var name = "xiaoming";
  function set(newName) {
    name = newName;
  }

  window.module1 = { set };
})(window, dependModule);

# AMD(浏览器)
